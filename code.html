                  // string traversal with assignment to array to reverse
                  function reverseString(argString) {
                    let length = argString.length;
                    let revString = [];
                    for (let index = length - 1; index >= 0; index--) {
                      revString += argString[index];
                    }
                    return revString;
                  }
reverseString(argString)

Reverses given argument utilizing a decremented for loop and assignment to an array 

                
                  
                  function isPalindrome(argString, argRevString) {
                    let returnObj = { reversed: argRevString };
                    if (argString == argRevString) {
                      returnObj.isPalindrome = true;
                      returnObj.msg = "It is a palindrome.";
                    } else {
                      returnObj.isPalindrome = false;
                      returnObj.msg = "It is not a palindrome.";
                    }
                    return returnObj;
                  }
isPalindrome(argString, argString)

Checks for two given arguments to be equal, one of the arguments being reversed, and returns an object containing the reversed string, a boolean showing if the word is a palindrome, and the message reflecting the boolean state
